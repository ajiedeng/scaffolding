
用法1：（大图标,开关个数为1）

    var outlets = [
            {name:"开关1",state:"关"},
        ];
    <SocketTop outlets={outlets}/>
    
用法2：（中图标,开关个数为2或3）

    var outlets = [
            {name:"开关1",state:"关"},
            {name:"开关2",state:"关"},
        ];
    <SocketTop outlets={outlets}/>
    
用法3：（小图标,开关个数大于3）

    var outlets = [
            {name:"开关1",state:"关"},
            {name:"开关2",state:"关"},
            {name:"开关3",state:"关"},
            {name:"开关4",state:"关"},
            {name:"开关5",state:"关"},
            {name:"开关6",state:"关"},
        ];
    <SocketTop outlets={outlets}/>
